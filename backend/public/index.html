Pengenalan JavaScript

JavaScript adalah bahasa pemrograman serbaguna yang menggerakkan interaktivitas web. Dalam tutorial ini, Anda akan memulai dari dasar dan maju ke topik tingkat ahli.

Apa itu JavaScript?

JavaScript adalah bahasa pemrograman yang digunakan untuk membuat halaman web interaktif. Dengan JavaScript, Anda dapat:

  • Mengubah konten HTML dan CSS
  • Merespons tindakan pengguna
  • Mengirim dan menerima data dari server
  • Membuat animasi dan efek visual

Mengapa Belajar JavaScript?

  • Populer: Salah satu bahasa pemrograman paling populer
  • Versatile: Dapat digunakan untuk front-end dan back-end
  • Mudah dimulai: Tidak memerlukan setup khusus
  • Banyak sumber belajar: Tutorial dan komunitas yang luas

Design Patterns


    // Singleton
    const Singleton = (function() {
      let instance;
      function createInstance() {
        return new Object("I am the instance");
      }
      return {
        getInstance: function() {
          if (!instance) {
            instance = createInstance();
          }
          return instance;
        }
      };
    })();

    console.log(Singleton.getInstance());

Advanced Asynchronous Patterns


    function timeout(ms) {
      return new Promise(resolve => setTimeout(resolve, ms));
    }

    async function* numberGenerator() {
      yield 1;
      await timeout(1000);
      yield 2;
      await timeout(1000);
      yield 3;
    }

    (async function() {
      for await (let num of numberGenerator()) {
        console.log(num);
      }
    })();

Prototypes and Inheritance


    function Vehicle(name) {
      this.name = name;
    }

    Vehicle.prototype.start = function() {
      return `${this.name} is starting`;
    };

    function Car(name) {
      Vehicle.call(this, name);
    }

    Car.prototype = Object.create(Vehicle.prototype);
    Car.prototype.constructor = Car;

    let myCar = new Car("Toyota");
    console.log(myCar.start());

Advanced Performance Optimization


    const memoize = (fn) => {
      const cache = new Map();
      return (...args) => {
        const key = JSON.stringify(args);
        if (cache.has(key)) return cache.get(key);
        const result = fn.apply(this, args);
        cache.set(key, result);
        return result;
      };
    };

    const expensiveFunction = memoize((n) => {
      console.log("Calculating...");
      return n * 2;
    });

    console.log(expensiveFunction(5)); // Calculating...
    console.log(expensiveFunction(5)); // Uses cached result

Metaprogramming and Reflection


    const handler = {
      get: function(target, name) {
        return name in target ? target[name] : 42;
      }
    };
    const p = new Proxy({}, handler);

Abstract Syntax Tree


    function parseExpression(expression) {
      const tokens = expression.split(' ');
      return {
        type: 'BinaryExpression',
        operator: tokens[1],
        left: { type: 'Literal', value: parseInt(tokens[0]) },
        right: { type: 'Literal', value: parseInt(tokens[2]) }
      };
    }

    console.log(parseExpression('5 + 3'));

Machine Learning with TensorFlow.js


    async function runModel() {
      const model = tf.sequential();
      model.add(tf.layers.dense({units: 1, inputShape: [1]}));
      model.compile({loss: 'meanSquaredError', optimizer: 'sgd'});
      
      const xs = tf.tensor2d([1, 2, 3, 4], [4, 1]);
      const ys = tf.tensor2d([1, 3, 5, 7], [4, 1]);
      
      await model.fit(xs, ys, {epochs: 10});
      console.log(model.predict(tf.tensor2d([5], [1, 1])));
    }

    runModel();
